my $salt='';
my $encrypted='';
my $password='';
my $use = 'Usage: Please provide password for encrypt';
my @saltchars=('.', '/', 0..9, 'A'..'Z', 'a'..'z');
my $args=@ARGV;
if ( $args< 1 || $args> 2 )
{
print "$use\n";
exit;
}
$password=$ARGV[0];
if( $args == 1 )
{
$salt = join('',@saltchars[rand(64),rand(64)]);
}
else
{
$salt=$ARGV[1];
}
$encrypted=crypt($password,$salt);
print "$password --> $encrypted\n";
